原始设计

短期记忆：Redis List 存储最近 10 条对话
每次更新使用 Lua 脚本执行 RPUSH + LTRIM，确保原子性滑动
当前实现

    redisTemplate.opsForList().rightPush(key, message);
    // ❌ 没有 LTRIM，导致历史记录无限增长！

问题影响
text

第1轮：存1条  → Redis: [msg1]
第2轮：存2条  → Redis: [msg1, msg2]
...
第20轮：存20条 → Redis: [msg1, msg2, ..., msg20]  ❌ 应该只保留10条！

结果：
1. 传给LLM的历史越来越长，Token消耗增加
2. buildCompletePrompt 中只取最近5个问题，但Redis里存了20个
3. 导致LLM看不到完整的已问问题列表！
修复方案
方案1：使用 Lua 脚本（原设计方案）

Java

// RedisMemoryService.java

/**
 * 使用 Lua 脚本原子性地添加消息并保持滑动窗口
 */
public void addMessage(String sessionId, String message) {
    String key = "interview:history:" + sessionId;
    
    // Lua 脚本：RPUSH + LTRIM 原子操作
    String luaScript = 
        "redis.call('RPUSH', KEYS[1], ARGV[1]) " +
        "redis.call('LTRIM', KEYS[1], -10, -1) " +  // 保留最近10条
        "return redis.call('LLEN', KEYS[1])";
    
    RedisScript<Long> script = RedisScript.of(luaScript, Long.class);
    Long size = redisTemplate.execute(
        script,
        Collections.singletonList(key),
        message
    );
    
    log.debug("消息已添加，当前历史记录数量: {}, sessionId: {}", size, sessionId);
    
    // 设置过期时间
    redisTemplate.expire(key, 2, TimeUnit.HOURS);
}
方案2：使用 Spring Redis Pipeline（性能更好）

Java

public void addMessage(String sessionId, String message) {
    String key = "interview:history:" + sessionId;
    
    redisTemplate.executePipelined(new SessionCallback<Object>() {
        @Override
        public Object execute(RedisOperations operations) throws DataAccessException {
            operations.opsForList().rightPush(key, message);
            operations.opsForList().trim(key, -10, -1); // 保留最近10条
            operations.expire(key, 2, TimeUnit.HOURS);
            return null;
        }
    });
}

// LLMService.java

public class LLMService {
    
    /**
     * 注册 Function Calling 工具
     */
    private ChatOptions buildChatOptionsWithTools() {
        // 定义工具：搜索标准答案
        FunctionCallback searchStandardAnswer = FunctionCallback.builder()
            .function("searchStandardAnswer", this::searchStandardAnswer)
            .description("搜索技术问题的标准答案，用于验证候选人回答的准确性")
            .inputType(SearchStandardAnswerRequest.class)
            .build();
        
        return ChatOptions.builder()
            .functions(List.of(searchStandardAnswer))
            .build();
    }
    
    /**
     * 工具实现：搜索标准答案
     */
    public String searchStandardAnswer(SearchStandardAnswerRequest request) {
        // 1. 从知识库查询（可以是向量数据库、Elasticsearch等）
        // 2. 或调用外部API（如技术文档、Stack Overflow等）
        
        log.info("调用工具：searchStandardAnswer, 查询: {}", request.getQuery());
        
        // 示例：从预定义知识库查询
        Map<String, String> knowledgeBase = Map.of(
            "Redis持久化", "Redis支持RDB和AOF两种持久化方式...",
            "Spring事务传播", "Spring定义了7种事务传播行为：REQUIRED, REQUIRES_NEW..."
        );
        
        return knowledgeBase.getOrDefault(request.getQuery(), "未找到相关标准答案");
    }
    
    @Data
    public static class SearchStandardAnswerRequest {
        private String query;
        private String category; // technical, behavioral, project
    }
}

问题
性能开销大：每次用户输入都调用 LLM
Token 消耗：增加成本
延迟高：影响用户体验
改进方案：使用规则 + 轻量级模型
Java

@Service
public class IntentDetectionService {
    
    /**
     * 轻量级意图检测（规则优先，LLM兜底）
     */
    public IntentResult detectIntent(String userInput, String currentStage) {
        
        // 1️⃣ 规则匹配（最快，0 Token）
        IntentResult ruleResult = detectByRules(userInput);
        if (ruleResult != null) {
            log.debug("规则匹配到意图: {}", ruleResult.getIntentType());
            return ruleResult;
        }
        
        // 2️⃣ 关键词匹配
        IntentResult keywordResult = detectByKeywords(userInput, currentStage);
        if (keywordResult != null) {
            log.debug("关键词匹配到意图: {}", keywordResult.getIntentType());
            return keywordResult;
        }
        
        // 3️⃣ LLM 兜底（只在必要时调用）
        log.debug("使用 LLM 进行意图检测");
        return detectByLLM(userInput, currentStage);
    }
    
    /**
     * 规则匹配
     */
    private IntentResult detectByRules(String userInput) {
        String normalized = userInput.toLowerCase().trim();
        
        // 薪资问题
        if (normalized.matches(".*(薪资|工资|薪水|待遇|年薪|月薪|offer).*")) {
            return new IntentResult(
                IntentType.COMPENSATION_QUESTION,
                "检测到薪资相关问题",
                false,
                "薪资相关问题我们会在后续阶段讨论，现在先聊聊技术方面的内容。"
            );
        }
        
        // 明确结束
        if (normalized.matches(".*(结束|不面|不想面|到此为止).*")) {
            return new IntentResult(
                IntentType.END_INTERVIEW,
                "用户明确要求结束",
                false,
                null
            );
        }
        
        // 攻击性言论
        if (normalized.matches(".*(傻|垃圾|智障|cnm|sb).*")) {
            return new IntentResult(
                IntentType.OFF_TOPIC,
                "检测到不当言论",
                true,
                "让我们保持专业，继续技术讨论。"
            );
        }
        
        return null; // 未匹配到规则
    }
    
    /**
     * 关键词匹配
     */
    private IntentResult detectByKeywords(String userInput, String currentStage) {
        // 在 OPENING 阶段，包含"我"、"毕业"、"项目"等关键词
        if ("OPENING".equals(currentStage)) {
            if (userInput.matches(".*(我|我的|我是|我叫|我来自|我学|我做|项目|毕业).*")) {
                return new IntentResult(
                    IntentType.VALID_TECHNICAL_QUESTION,
                    "OPENING阶段的自我介绍",
                    true,
                    null
                );
            }
        }
        
        return null;
    }
    
    /**
     * LLM 检测（兜底）
     */
    private IntentResult detectByLLM(String userInput, String currentStage) {
        String prompt = buildIntentDetectionPrompt(userInput, currentStage);
        String result = llmService.generateResponse(prompt, "", new ArrayList<>());
        return parseIntentResult(result);
    }
}
